$("#source_button_slide").on("click", function() {
$("#source_slide").removeClass("hidden")
$("#source_slide").addClass("visible")
$("#x_slide").removeClass("hidden")
$("#x_slide").addClass("visible")
})
$("#fifth_element").on("click", function() {
if($("#fifth_element").hasClass("fifth_element_position2")) {
$("#fifth_element").removeClass("fifth_element_position2")
$("#fifth_element").addClass("fifth_element_position1")
} else {
$("#fifth_element").removeClass("fifth_element_position1")
$("#fifth_element").addClass("fifth_element_position2")
}
})

$("#x_slide").on("click", function() {
if($("#x_slide").hasClass("visible")){
$("#x_slide").removeClass("visible")
$("#x_slide").addClass("hidden")
$("#source_slide").removeClass("visible")
$("#source_slide").addClass("hidden")
}})
